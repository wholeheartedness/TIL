**â—Solution 5-1 : Switch-case**

```java
public class Solution_p181926 {
    public int solution(int n, String control) {
        char key;
        char[] chars = control.toCharArray();
        for (int i = 0; i < control.length(); i++) {
            key = chars[i];
            switch (key) {
                case 'w':
                    n += 1;
                    break;
                case 's':
                    n -= 1;
                    break;
                case 'a':
                    n -= 10;
                    break;
                case 'd':
                    n += 10;
                    break;
            }
        }
        return n;
    }

    public static void main(String[] args) {
        System.out.println(new Solution_p181926().solution(0, "wsdawsdassw"));

    }
}
```
